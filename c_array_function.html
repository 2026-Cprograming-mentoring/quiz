CHAPTER 01

배열이란? — 같은 타입의 데이터 묶음

학생 30명의 점수를 저장해야 한다면 변수를 30개 만들어야 할까요? 아닙니다. 배열(Array)을 쓰면 같은 타입의 데이터를 하나의 이름으로 묶어서 관리할 수 있어요.

사물함이 일렬로 늘어선 모습을 떠올려보세요. 각 칸에 번호(인덱스)가 있고, 그 안에 값이 들어 있어요. 배열도 정확히 이런 구조입니다.

💡
배열의 3가지 핵심 특징
① 같은 타입의 데이터만 저장 가능
② 메모리에 연속으로 저장됨
③ 인덱스는 항상 0부터 시작
⚠️
크기가 5인 배열의 인덱스는 0, 1, 2, 3, 4예요. 5가 아닙니다! 이 실수가 가장 많아요.

int scores[5] = {95, 82, 74, 91, 88};

[0]95
[1]82
[2]74
[3]91
[4]88

인덱스 [0]부터 시작 — 마지막 인덱스는 [크기-1]

CHAPTER 02

배열 선언과 초기화 — 다양한 방법

선언 & 초기화 방법들
// 방법 1: 선언과 동시에 초기화
int scores[5] = {95, 82, 74, 91, 88};

// 방법 2: 크기 생략 (컴파일러가 자동 계산)
int nums[] = {10, 20, 30}; // 크기 3

// 방법 3: 먼저 선언, 나중에 값 넣기
int ages[3];
ages[0] = 20;
ages[1] = 22;
ages[2] = 21;

// 방법 4: 전부 0으로 초기화
int zeros[100] = {0};

// 방법 5: 일부만 초기화 (나머지는 0)
int partial[5] = {1, 2};
// → {1, 2, 0, 0, 0}

선언 문법

타입 이름[크기]; 배열 선언의 기본 형태
타입: int, float, double, char 등
이름: 변수 이름과 같은 규칙
크기: 반드시 상수여야 함 (변수 불가!)
🚨
초기화 없이 선언만 하면 쓰레기 값이 들어 있어요! 반드시 초기화하는 습관을 들이세요.

배열 원소 읽기 & 쓰기

access.c
int scores[] = {95, 82, 74, 91, 88};

// 읽기
printf("%d\n", scores[0]);  // 95
printf("%d\n", scores[3]);  // 91

// 쓰기 (값 변경)
scores[2] = 80;  // 74 → 80

// ❌ 위험! 범위 초과
scores[5] = 100;  // 크기 5인데 [5]는 없음!
scores[-1] = 0;   // 음수 인덱스도 불가!
🚨
배열 범위 초과(Out of Bounds)는 C언어에서 가장 위험한 버그예요. 컴파일러가 경고도 안 해주고, 실행하면 엉뚱한 메모리를 건드려서 프로그램이 크래시하거나 예측 불가능한 동작을 합니다. 항상 인덱스가 0 이상, 크기 미만인지 확인하세요.

배열 크기 구하는 공식

int size = sizeof(arr) / sizeof(arr[0]);

배열 전체 바이트 ÷ 원소 하나 바이트 = 원소 개수

CHAPTER 03

배열과 메모리 — 연속된 공간

배열의 원소들은 메모리에 빈틈 없이 연속으로 저장돼요. int가 4바이트이므로, 시작 주소가 1000이면 다음 원소는 1004, 그 다음은 1008에 저장됩니다.

int arr[4] = {10, 20, 30, 40};

[0]101000
[1]201004
[2]301008
[3]401012

각 칸 = 4바이트 (int 크기) · 주소가 4씩 증가

address.c
#include <stdio.h>

int main() {
    int arr[4] = {10, 20, 30, 40};

    for (int i = 0; i < 4; i++) {
        printf("arr[%d] = %d (주소: %p)\n",
               i, arr[i], &arr[i]);
    }

    // 배열 이름 = 첫 번째 원소의 주소
    printf("arr = %p\n", arr);
    printf("&arr[0] = %p\n", &arr[0]);
    // 두 값이 동일!

    return 0;
}
💡
배열 이름 = 주소! 배열 이름 자체가 첫 번째 원소의 주소예요. 나중에 함수에 배열을 넘길 때 이 개념이 핵심이 됩니다.

CHAPTER 04

배열 순회 — for문과 함께

traverse.c
#include <stdio.h>

int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int size = sizeof(arr) / sizeof(arr[0]);

    // 전체 출력
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    // 출력: 10 20 30 40 50

    // 역순 출력
    for (int i = size - 1; i >= 0; i--) {
        printf("%d ", arr[i]);
    }
    // 출력: 50 40 30 20 10

    return 0;
}

순회 패턴 외우기

for (i = 0; i < size; i++) 정방향 순회
for (i = size-1; i >= 0; i--) 역방향 순회
💡
핵심 포인트: 조건식에서 i < size를 쓰세요. i <= size로 쓰면 범위 초과!
📌
사용자 입력 받기: scanf와 for문을 조합하면 여러 값을 입력받아 배열에 저장할 수 있어요.
for(i=0; i<n; i++) scanf("%d", &arr[i]);
CHAPTER 05

배열 활용 패턴 — 합계 / 최대 / 검색 / 정렬

📊

합계 & 평균

▼
sum_avg.c
int scores[] = {85, 92, 78, 96, 88};
int n = 5;

// 합계
int sum = 0;
for (int i = 0; i < n; i++) {
    sum += scores[i];
}
printf("합계: %d\n", sum);  // 439

// 평균 (형변환 필수!)
double avg = (double)sum / n;
printf("평균: %.1f\n", avg); // 87.8
⚠️
(double) 캐스팅을 빠뜨리면 정수 나눗셈이 되어 소수점이 잘려요! 439 / 5 = 87이 되어버립니다. 반드시 (double)로 형변환하세요.
💡
패턴: 누적 변수 sum을 0으로 초기화 → for문으로 하나씩 더하기. 배열 문제의 80%는 이 패턴이에요!
🏆

최댓값 & 최솟값

▼
min_max.c
int arr[] = {34, 12, 56, 7, 89, 23};
int n = 6;

int max = arr[0];  // 첫 원소로 초기화!
int min = arr[0];

for (int i = 1; i < n; i++) {
    if (arr[i] > max) max = arr[i];
    if (arr[i] < min) min = arr[i];
}
printf("최대: %d, 최소: %d\n", max, min);
// 최대: 89, 최소: 7
💡
초기값을 arr[0]으로! 0이나 99999 같은 임의의 값으로 초기화하면 실수하기 쉬워요. 첫 번째 원소로 초기화하고, 비교는 인덱스 1부터 시작하세요.
🔍

선형 검색 (Linear Search)

▼
search.c
int arr[] = {15, 42, 8, 73, 29};
int n = 5;
int target = 73;
int found = -1;  // -1 = 못 찾음

for (int i = 0; i < n; i++) {
    if (arr[i] == target) {
        found = i;
        break;  // 찾으면 즉시 종료
    }
}

if (found != -1)
    printf("인덱스 %d에서 발견\n", found);
else
    printf("값을 찾지 못함\n");

처음부터 끝까지 하나씩 비교하면서 찾는 가장 단순한 검색 방법이에요. 찾으면 break로 빠져나오는 것이 포인트!

📌
-1을 쓰는 이유: 배열 인덱스는 0 이상이므로, -1은 "유효한 인덱스가 아니다" = "못 찾았다"는 의미로 사용해요.
📈

버블 정렬 (Bubble Sort)

▼

옆에 있는 두 원소를 비교해서 순서가 잘못되면 교환(swap)하는 과정을 반복합니다. 가장 큰 값이 거품처럼 뒤로 밀려나가서 '버블' 정렬이라 불려요.

bubble_sort.c
int arr[] = {64, 25, 12, 22, 11};
int n = 5;

for (int i = 0; i < n - 1; i++) {
    for (int j = 0; j < n - i - 1; j++) {
        if (arr[j] > arr[j + 1]) {
            // swap (두 값 교환)
            int temp = arr[j];
            arr[j] = arr[j + 1];
            arr[j + 1] = temp;
        }
    }
}
// 결과: 11 12 22 25 64
💡
swap의 3단계: ① temp에 a 저장 → ② a에 b 넣기 → ③ b에 temp 넣기. 임시 변수 없이는 교환할 수 없어요!
🔤

문자 배열 (문자열)

▼

C언어에서 문자열은 char 배열이에요. 끝에 '\0'(널 문자)이 자동으로 들어가서 문자열의 끝을 표시합니다.

string.c
// 방법 1: 문자열 리터럴로 초기화
char name[] = "Hello";
// 내부: {'H','e','l','l','o','\0'} → 크기 6!

// 방법 2: 한 글자씩 초기화
char greet[6] = {'H','e','l','l','o','\0'};

// 출력
printf("%s\n", name);    // Hello
printf("%c\n", name[0]); // H
printf("%c\n", name[4]); // o

// 글자 수 (strlen - string.h 필요)
printf("%lu\n", strlen(name)); // 5 (널 문자 미포함)
⚠️
"Hello"는 글자 5개지만 배열 크기는 6! 끝에 '\0'이 들어가기 때문이에요. 배열 크기를 잡을 때 글자 수 + 1을 해야 합니다.
CHAPTER 06

2차원 배열 — 행과 열

2차원 배열은 배열 안에 배열이 있는 구조예요. 엑셀 시트처럼 행(row)과 열(column)로 데이터를 정리할 수 있어요.

int m[2][3] = { {1,2,3}, {4,5,6} };

행 0
[0][0]1
[0][1]2
[0][2]3
행 1
[1][0]4
[1][1]5
[1][2]6

접근: 배열[행][열] — m[1][2] = 6

2d_array.c
int matrix[2][3] = {
    {1, 2, 3},
    {4, 5, 6}
};

// 이중 for문으로 순회
for (int i = 0; i < 2; i++) {       // 행
    for (int j = 0; j < 3; j++) {   // 열
        printf("%d ", matrix[i][j]);
    }
    printf("\n");
}
// 출력:
// 1 2 3
// 4 5 6

// 개별 접근
matrix[1][2] = 99; // 6 → 99
📌
2차원 = 이중 for문! 바깥 루프가 행을, 안쪽 루프가 열을 순회합니다. 선언할 때 열의 크기는 반드시 명시해야 해요. 행은 생략 가능하지만 열은 안 됩니다.

함수란? — 정의 · 호출 · 반환

같은 코드를 여러 번 반복하는 대신, 한 번 만들어놓고 이름으로 호출하는 구조예요. 코드의 재사용성과 가독성을 모두 높여줍니다.

반환타입 함수이름(매개변수) {
    // 실행할 코드
    return 반환값;
}
반환타입

돌려주는 값의 타입. 없으면 void

매개변수

외부에서 받는 입력값. 여러 개 가능

function.c
#include <stdio.h>

// ① 반환값 있는 함수
int add(int a, int b) {
    return a + b;
}

// ② 반환값 없는 함수 (void)
void greet(char name[]) {
    printf("안녕, %s!\n", name);
}

// ③ 매개변수 없는 함수
void printLine() {
    printf("----------\n");
}

int main() {
    int result = add(3, 5);
    printf("%d\n", result); // 8

    greet("철수");
    printLine();
    printf("%d\n", add(10, 20)); // 30

    return 0;
}
⚠️
함수 위치 주의! C에서는 함수를 호출하기 전에 정의하거나, 위에 함수 원형(프로토타입)을 선언해야 해요. int add(int a, int b); — 끝에 세미콜론!
CHAPTER 08

매개변수 전달 — 값 전달 vs 포인터 전달

📦 값 전달 (Call by Value)

값을 복사해서 넘기므로, 함수 안에서 바꿔도 원본은 그대로예요.

값 전달
void change(int x) {
    x = 100;  // 복사본만 변경됨
}

int main() {
    int num = 5;
    change(num);
    printf("%d", num); // 여전히 5!
}

📍 포인터 전달 (Call by Reference)

주소를 넘기므로, 함수 안에서 바꾸면 원본도 변경됩니다.

포인터 전달
void change(int *x) {  // 포인터로 받기
    *x = 100;  // 원본 직접 변경
}

int main() {
    int num = 5;
    change(&num);  // 주소 넘기기
    printf("%d", num); // 100!
}

실전 예: 두 값 교환 (swap)

swap.c
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x = 10, y = 20;
    swap(&x, &y);
    printf("%d, %d", x, y); // 20, 10
}
💡
언제 포인터 전달? ① 함수 안에서 원본을 수정해야 할 때 (swap 등), ② scanf가 &를 붙이는 이유도 같아요 — scanf가 변수에 값을 "넣어줘야" 하니까 주소를 알아야 합니다.
CHAPTER 09

함수 + 배열 — 배열을 함수에 넘기기

배열을 함수에 넘기면 주소만 전달돼요. 복사가 아니라 원본을 직접 가리키는 거예요. 그래서 함수 안에서 배열을 수정하면 원본도 바뀝니다.

또한 함수 안에서는 배열 크기를 알 수 없으므로, 크기를 별도 매개변수로 함께 전달하는 게 관례예요.

🚨
배열은 항상 주소 전달! 포인터(*)나 &를 안 써도 그래요. 배열 이름 자체가 주소이기 때문이에요.
일반 변수배열
전달 방식값 복사주소 전달
원본 변경안 됨항상 가능
크기 정보불필요별도 전달
array_func.c
#include <stdio.h>

// 배열 출력
void printArr(int arr[], int n) {
    for (int i = 0; i < n; i++)
        printf("%d ", arr[i]);
    printf("\n");
}

// 배열 합계
int getSum(int arr[], int n) {
    int sum = 0;
    for (int i = 0; i < n; i++)
        sum += arr[i];
    return sum;
}

// 배열 전체 2배 (원본 수정!)
void doubleAll(int arr[], int n) {
    for (int i = 0; i < n; i++)
        arr[i] *= 2;
}

int main() {
    int data[] = {10, 25, 8, 47};
    int n = 4;

    printArr(data, n);   // 10 25 8 47
    printf("합: %d\n", getSum(data, n));

    doubleAll(data, n);
    printArr(data, n);   // 20 50 16 94
    return 0;
}
